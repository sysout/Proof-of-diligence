        int ans = Integer.MAX_VALUE;
        for (int lo = 0, hi = 0, sum = 0; lo < nums.length; lo++){

                sum += nums[hi++];
            }
            if (sum >= s){

      public int encPerm(int[] a){
          int n = a.length;
          int used = 0;
          int ret = 0;
          for(int i = 0;i < n;i++){

          }

            while (lo <=hi && A[lo] < piv){
                lo++;
            }

                hi--;
